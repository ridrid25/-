## Apps Script код

```javascript
function updateStatus(manager,status){
  const ss=SpreadsheetApp.getActiveSpreadsheet();
  const sheet=ss.getSheetByName('Данные');
  const dataRange=sheet.getRange(2,2,sheet.getLastRow()-1,25);
  const values=dataRange.getValues();
  const allowed=['Новый','Обещал оплатить','Частичная оплата','Оплатил','Недоступен'];

  if(allowed.indexOf(status)===-1){
    throw new Error('Недопустимый статус');
  }

  for(let i=0;i<values.length;i++){
    if(values[i][0]===manager){
      sheet.getRange(i+2,25).setValue(status);
      logAction(manager,'Обновлен статус: '+status);
      return true;
    }
  }

  throw new Error('Менеджер не найден');
}

function logAction(manager,action){
  const ss=SpreadsheetApp.getActiveSpreadsheet();
  let logSheet=ss.getSheetByName('Лог');

  if(!logSheet){
    logSheet=ss.insertSheet('Лог');
    logSheet.getRange(1,1,1,3).setValues([['Дата','Менеджер','Действие']]);
  }

  logSheet.appendRow([new Date(),manager,action]);
}
```
